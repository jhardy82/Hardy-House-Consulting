```python
from pptx import Presentation
from pptx.util import Inches, Pt, Emu
from pptx.dml.color import RGBColor
from pptx.enum.text import PP_ALIGN
from pptx.util import Inches, Pt
import os

# Colours
VOID     = RGBColor(0x07, 0x04, 0x0F)
GOLD     = RGBColor(0xC4, 0x9A, 0x1F)
AETHER   = RGBColor(0x9B, 0x7B, 0xE0)
WHITE    = RGBColor(0xF4, 0xF0, 0xEB)
DIM      = RGBColor(0x80, 0x78, 0x70)
FIRE     = RGBColor(0xC4, 0x9A, 0x1F)
EARTH    = RGBColor(0x2D, 0x80, 0x50)
AIR      = RGBColor(0x3B, 0x5F, 0xC8)
WATER    = RGBColor(0x20, 0xA8, 0xC8)
ROYAL    = RGBColor(0x1E, 0x3F, 0xAA)
FOREST   = RGBColor(0x1B, 0x5E, 0x35)

# Slide dimensions: 33.87cm x 19.05cm (widescreen 16:9)
W = Inches(13.33)
H = Inches(7.5)

prs = Presentation()
prs.slide_width  = W
prs.slide_height = H

BLANK = prs.slide_layouts[6]  # blank layout

def add_slide():
    return prs.slides.add_slide(BLANK)

def bg(slide):
    """Fill slide background with void black."""
    fill = slide.background.fill
    fill.solid()
    fill.fore_color.rgb = VOID

def txt(slide, text, left, top, width, height,
        size=28, bold=False, italic=False,
        color=WHITE, align=PP_ALIGN.LEFT, font='Georgia'):
    txBox = slide.shapes.add_textbox(left, top, width, height)
    tf    = txBox.text_frame
    tf.word_wrap = True
    p  = tf.paragraphs[0]
    p.alignment = align
    run = p.add_run()
    run.text = text
    run.font.size    = Pt(size)
    run.font.bold    = bold
    run.font.italic  = italic
    run.font.color.rgb = color
    run.font.name    = font
    return txBox

def eyebrow(slide, label, top=Inches(1.8)):
    txt(slide, label.upper(), Inches(0.9), top, Inches(6), Inches(0.4),
        size=9, color=GOLD, font='Courier New')

def accent_line(slide, top=Inches(2.1)):
    line = slide.shapes.add_shape(1, Inches(0.9), top, Inches(1.2), Emu(12700))
    line.fill.solid(); line.fill.fore_color.rgb = GOLD
    line.line.fill.background()

def solid_placeholder(slide, label, color, right=True):
    """Add a simple geometric shape as Platonic solid placeholder."""
    cx = Inches(9.5) if right else Inches(1)
    cy = Inches(2.5)
    sz = Inches(3.2)
    shape = slide.shapes.add_shape(9, cx, cy, sz, sz)  # hexagon as stand-in
    shape.fill.solid(); shape.fill.fore_color.rgb = RGBColor(
        int(color.rgb[0]*0.15), int(color.rgb[1]*0.15), int(color.rgb[2]*0.15))
    shape.line.color.rgb = color
    shape.line.width = Pt(1.5)
    cap = slide.shapes.add_textbox(cx + Inches(0.8), cy + sz + Inches(0.1),
                                    Inches(1.8), Inches(0.4))
    cp = cap.text_frame.paragraphs[0]
    cp.alignment = PP_ALIGN.CENTER
    r = cp.add_run(); r.text = label
    r.font.size = Pt(10); r.font.color.rgb = color; r.font.name = 'Courier New'

# ── Slide 1: Cover ────────────────────────────────────────────────
s1 = add_slide(); bg(s1)
solid_placeholder(s1, 'Dodecahedron', AETHER)
txt(s1, 'Hardy House', Inches(0.9), Inches(2.0), Inches(7), Inches(1.6),
    size=60, bold=True, color=WHITE, font='Georgia')
txt(s1, 'Consulting', Inches(0.9), Inches(3.3), Inches(7), Inches(1.0),
    size=40, italic=True, color=GOLD, font='Georgia')
txt(s1, 'Structured Intelligence for Complex Systems',
    Inches(0.9), Inches(4.5), Inches(8), Inches(0.6),
    size=16, color=DIM, font='Times New Roman')
txt(s1, 'ContextForge', Inches(0.9), Inches(6.5), Inches(4), Inches(0.5),
    size=10, color=AETHER, font='Courier New')

# ── Slide 2: The Problem ──────────────────────────────────────────
s2 = add_slide(); bg(s2)
eyebrow(s2, 'The Challenge')
accent_line(s2)
txt(s2, 'Complexity grows.\nCoordination fails.\nDelivery slows.',
    Inches(0.9), Inches(2.2), Inches(8), Inches(2.5),
    size=36, color=WHITE, font='Georgia')
for i, bullet in enumerate([
    'Teams scale faster than the systems that coordinate them',
    'Structure exists in silos — not as a shared operational language',
    'Delivery confidence degrades as complexity compounds',
]):
    txt(s2, f'— {bullet}', Inches(1.0), Inches(4.6 + i*0.65), Inches(10), Inches(0.6),
        size=14, color=DIM, font='Times New Roman')

# ── Slide 3: Our Approach ─────────────────────────────────────────
s3 = add_slide(); bg(s3)
eyebrow(s3, 'The Hardy House Approach')
accent_line(s3)
txt(s3, 'ContextForge', Inches(0.9), Inches(2.2), Inches(11), Inches(1.0),
    size=42, bold=True, color=GOLD, font='Georgia')
txt(s3, 'Geometry-backed agent coordination for complex systems delivery.',
    Inches(0.9), Inches(3.3), Inches(9), Inches(0.7),
    size=20, italic=True, color=WHITE, font='Georgia')
txt(s3, 'We assign the five classical elements to agent roles, apply sacred geometry\nas the structural logic, and run every engagement through the PAOAL cycle\n(Plan → Act → Observe → Assess → Learn) to build self-reinforcing delivery systems.',
    Inches(0.9), Inches(4.2), Inches(10), Inches(2.0),
    size=14, color=DIM, font='Times New Roman')

# ── Slide 4: The Five Elements ────────────────────────────────────
s4 = add_slide(); bg(s4)
eyebrow(s4, 'The Element System')
accent_line(s4)
txt(s4, 'Five Elements.\nFive Roles.',
    Inches(0.9), Inches(2.2), Inches(5.5), Inches(1.5),
    size=36, color=WHITE, font='Georgia')
elements = [
    ('Fire',   'Tetrahedron',   'Initiator', FIRE),
    ('Earth',  'Hexahedron',    'Builder',   EARTH),
    ('Air',    'Octahedron',    'Orchestrator', AIR),
    ('Water',  'Icosahedron',   'Connector', WATER),
    ('Aether', 'Dodecahedron',  'Framework', AETHER),
]
for i, (el, solid, role, col) in enumerate(elements):
    x = Inches(0.9 + i * 2.35)
    shape = s4.shapes.add_shape(1, x, Inches(4.0), Inches(1.8), Inches(1.8))
    shape.fill.solid(); shape.fill.fore_color.rgb = RGBColor(
        max(0, col.rgb[0]//6), max(0, col.rgb[1]//6), max(0, col.rgb[2]//6))
    shape.line.color.rgb = col; shape.line.width = Pt(1)
    txt(s4, el,    x, Inches(5.95), Inches(2), Inches(0.4), size=12, bold=True, color=col, font='Georgia', align=PP_ALIGN.CENTER)
    txt(s4, solid, x, Inches(6.35), Inches(2), Inches(0.4), size=9, color=DIM, font='Courier New', align=PP_ALIGN.CENTER)
    txt(s4, role,  x, Inches(6.7),  Inches(2), Inches(0.4), size=9, italic=True, color=WHITE, font='Times New Roman', align=PP_ALIGN.CENTER)

# ── Slide 5: Sacred Geometry ──────────────────────────────────────
s5 = add_slide(); bg(s5)
eyebrow(s5, 'Sacred Geometry')
accent_line(s5)
txt(s5, 'Structure is\nIntelligence.',
    Inches(0.9), Inches(2.2), Inches(5.5), Inches(1.5),
    size=36, color=WHITE, font='Georgia')
for i, (label, val, desc) in enumerate([
    ('Flower of Life',  '19 pts',  'The source grid — all patterns emerge from here'),
    ('Fruit of Life',   '13 pts',  'The 13-circle selection (Metatron source)'),
    ("Metatron's Cube", '78 edges','C(13,2) — every possible connection, made explicit'),
]):
    y = Inches(4.0 + i*0.9)
    txt(s5, label, Inches(0.9), y, Inches(3.5), Inches(0.45), size=16, bold=True, color=GOLD, font='Georgia')
    txt(s5, val,   Inches(4.5), y, Inches(1.5), Inches(0.45), size=13, color=AETHER, font='Courier New')
    txt(s5, desc,  Inches(6.1), y, Inches(6.5), Inches(0.45), size=11, italic=True, color=DIM, font='Times New Roman')

# ── Slide 6: PAOAL Cycle ─────────────────────────────────────────
s6 = add_slide(); bg(s6)
eyebrow(s6, 'The PAOAL Cycle')
accent_line(s6)
txt(s6, 'Plan → Act → Observe\n→ Assess → Learn',
    Inches(0.9), Inches(2.2), Inches(7), Inches(1.8),
    size=34, color=WHITE, font='Georgia')
txt(s6, 'Maps to Torus geometry — a self-reinforcing loop.\nEvery engagement is a cycle, not a project.',
    Inches(0.9), Inches(4.2), Inches(9), Inches(1.0),
    size=16, italic=True, color=DIM, font='Times New Roman')
for i, (phase, col) in enumerate([('Plan',GOLD),('Act',FIRE),('Observe',AIR),('Assess',WATER),('Learn',AETHER)]):
    txt(s6, phase, Inches(0.9 + i*2.3), Inches(5.5), Inches(2.2), Inches(0.5),
        size=13, bold=True, color=col, font='Courier New', align=PP_ALIGN.CENTER)

# ── Slide 7: The Golden Ratio ─────────────────────────────────────
s7 = add_slide(); bg(s7)
eyebrow(s7, 'The Golden Ratio')
accent_line(s7)
txt(s7, 'φ = 1.618…', Inches(0.9), Inches(2.2), Inches(7), Inches(1.2),
    size=52, bold=True, color=GOLD, font='Georgia')
txt(s7, 'Each system level is φ× the one below.\nFibonacci team sizing: 2 · 3 · 5 · 8 · 13 · 21',
    Inches(0.9), Inches(3.6), Inches(9), Inches(1.2),
    size=18, color=WHITE, font='Georgia')
txt(s7, 'PHI governs hierarchy across ContextForge: agent counts, engagement phases,\ndocumentation depth, and review cycles are all calibrated to the golden ratio.',
    Inches(0.9), Inches(5.0), Inches(10), Inches(1.2),
    size=13, color=DIM, italic=True, font='Times New Roman')

# ── Slide 8: Tree of Life ─────────────────────────────────────────
s8 = add_slide(); bg(s8)
eyebrow(s8, 'Tree of Life')
accent_line(s8)
txt(s8, 'The Agent\nCoordination Chain.',
    Inches(0.9), Inches(2.2), Inches(6), Inches(1.6),
    size=34, color=WHITE, font='Georgia')
txt(s8, 'The Kabbalistic Tree of Life — 10 Sephirot, 22 paths — maps\ndirectly to the ContextForge agent coordination topology.\nEach node is an agent role. Each path is a communication channel.',
    Inches(0.9), Inches(4.1), Inches(10), Inches(1.6),
    size=14, color=DIM, italic=True, font='Times New Roman')
txt(s8, '10 nodes · 22 paths · 5 elemental assignments',
    Inches(0.9), Inches(6.0), Inches(10), Inches(0.6),

# ── Slide 9: Team Architecture ───────────────────────────────────
s9 = add_slide(); bg(s9)
eyebrow(s9, 'Team Architecture · GCMT')
accent_line(s9)
txt(s9, 'Geometry-backed\nTeam Design.',
    Inches(0.9), Inches(2.2), Inches(6), Inches(1.5),
    size=34, color=WHITE, font='Georgia')
shapes_list = [
    ('Triangle',     'Triad (3)',      'Adversarial review', GOLD),
    ('Tetrahedron',  'Pentad (5)',     'Foundational change', FIRE),

    ('Dodecahedron', 'Tridecad (13)', 'Enterprise scope',    AETHER),

for i, (shape, size, role, col) in enumerate(shapes_list):
    y = Inches(4.0 + i * 0.72)
    txt(s9, shape, Inches(0.9), y, Inches(2.5), Inches(0.55), size=14, bold=True, color=col, font='Georgia')
    txt(s9, size,  Inches(3.5), y, Inches(2.0), Inches(0.55), size=12, color=WHITE, font='Courier New')
    txt(s9, role,  Inches(5.6), y, Inches(6.0), Inches(0.55), size=12, color=DIM,   font='Times New Roman')

# ── Slide 10: Sri Yantra ─────────────────────────────────────────
s10 = add_slide(); bg(s10)
eyebrow(s10, 'Sri Yantra · PAOAL Phase Map')
accent_line(s10)
txt(s10, '9 Triangles.\n5 Phases.',
    Inches(0.9), Inches(2.2), Inches(5.5), Inches(1.5),
    size=36, color=WHITE, font='Georgia')
paoal_map = [
    ('1',   'Learn',   '#9B7BE0', AETHER),
    ('2–3', 'Assess',  '#1E3FAA', ROYAL),
    ('4–5', 'Observe', '#20A8C8', WATER),
    ('6–7', 'Act',     '#1B5E35', FOREST),
    ('8–9', 'Plan',    '#C49A1F', GOLD),

for i, (tri, phase, _, col) in enumerate(paoal_map):
    y = Inches(4.1 + i * 0.58)
    txt(s10, f'Triangles {tri}', Inches(0.9), y, Inches(2.5), Inches(0.5), size=11, color=DIM, font='Courier New')

s11 = add_slide(); bg(s11)
eyebrow(s11, 'Engagement Model')

for i, (phase, desc) in enumerate([

    ('Framework',  'Apply ContextForge — element assignment, PAOAL cadence, team geometry'),
    ('Delivery',   'Execute with structured intelligence — measurable, repeatable, auditable'),

    y = Inches(4.1 + i * 0.9)
    txt(s11, f'{i+1}. {phase}', Inches(0.9), y, Inches(3.0), Inches(0.55), size=16, bold=True, color=GOLD, font='Georgia')
    txt(s11, desc, Inches(3.9), y, Inches(8.5), Inches(0.55), size=13, color=DIM, italic=True, font='Times New Roman')

# ── Slide 12: Call to Action ──────────────────────────────────────
s12 = add_slide(); bg(s12)
txt(s12, 'Start the\nConversation.',
    Inches(0.9), Inches(2.0), Inches(9), Inches(2.0),

txt(s12, 'james.hardy1124@gmail.com',
    Inches(0.9), Inches(4.3), Inches(11.5), Inches(0.7),
    size=22, color=WHITE, font='Courier New', align=PP_ALIGN.CENTER)
txt(s12, 'Hardy House Consulting · ContextForge',
    Inches(0.9), Inches(5.5), Inches(11.5), Inches(0.6),

# ── Output ────────────────────────────────────────────────────────
out = os.path.join(os.path.dirname(__file__), '..', 'docs', 'hardy-house-deck.pptx')

prs.save(out)
print(f'Saved: {out}')
print(f'Slides: {len(prs.slides)}')
```
